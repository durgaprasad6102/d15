def is_prime(num):
    if num < 2:
        return False

    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False

    return True


def next_prime(n):
    num = n + 1

    while True:
        if is_prime(num):
            return num
        num += 1


n = int(input())
print(next_prime(n))
